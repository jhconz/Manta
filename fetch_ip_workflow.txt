# 2. Use this script to enable WiFi on all cameras:

from open_gopro import WirelessGoPro
import asyncio

async def enable_wifi_all(macs):
    for mac in macs:
        gopro = WirelessGoPro(target=mac)
        await gopro.open()
        await gopro.ble_command.enable_wifi_ap(enable=True)
        await gopro.close()
        print(f"Enabled WiFi on {mac}")

macs = ["AA:BB:CC:DD:EE:01", "AA:BB:CC:DD:EE:02", 
        "AA:BB:CC:DD:EE:03", "AA:BB:CC:DD:EE:04"]
asyncio.run(enable_wifi_all(macs))
